  \begin{itemize}

  \end{itemize}

\pause
  \item If some parameters could not be inferred, enumerate all admissible parametrizations, regarding:
  \begin{itemize}
    \item[--] Biological constraints \tcite{\cbccdmrcmmb}

  \end{itemize}
